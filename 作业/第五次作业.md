void backtrack(int i)
{
    if(i==n)
    {
        int ld = len(i);
        if(ld < minLen)
        {
            minLen = ld;
            for(int j=1; j<=n; j++)
                bestx[j] = x[j]; 
        }
    }
    else
        for(int r=i; r<=n; r++)
        {
            swap(x[i], x[r]);

                backtrack(i+1);

        }
}
